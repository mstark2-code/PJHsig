  function escapeHtml(str) {

    return String(str || "")

      .replaceAll("&", "&amp;")

      .replaceAll("<", "&lt;")

      .replaceAll(">", "&gt;")

      .replaceAll('"', "&quot;")

      .replaceAll("'", "&#039;");

  }


  function formatPhoneDisplay(str) {

    const digits = String(str || "").replace(/[^\d]/g, "");

    if (digits.length !== 10) return str;

    return digits.slice(0,3) + "." + digits.slice(3,6) + "." + digits.slice(6);

  }


  function phoneToTel(str) {

    const digits = String(str || "").replace(/[^\d]/g, "");

    if (digits.length === 10) return "1" + digits;

    return digits;

  }

    }


    const phoneDisplay = escapeHtml(formatPhoneDisplay(rawPhone));

    const phoneTel = phoneToTel(rawPhone);
